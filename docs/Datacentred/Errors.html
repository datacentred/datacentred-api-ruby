def self.raise_unless_successful(status, body)
  return if status.to_s.start_with? "2" # 2xx
  err = errors[status]
  message = body&.fetch("errors")&.first&.fetch("detail")
  if err
    raise err, message || status.to_s
  else
    raise Error, "Error #{status}: #{message}"
  end
end
